2. Crée un utilisateur manuellement :

>>
use App\Models\User;
use Illuminate\Support\Facades\Hash;

// création d'un user admin
$user = new User();
$user->name = 'admin';                   // username
$user->email = 'admin@gmail.com';        // email obligatoire
$user->password = Hash::make('password'); // mot de passe hashé
$user->email_verified_at = now();        // déjà vérifié
$user->role = 'admin';                   // rôle admin si tu utilises le champ role

$user->save();
